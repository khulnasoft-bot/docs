```kotlin Kotlin
import co.teleflow.Teleflow
import co.teleflow.dto.request.NotificationRequest
import co.teleflow.extensions.notifications

fun main() {
    val teleflow = Teleflow(apiKey = "<TELEFLOW_API_KEY>")
    val notificationRequest = NotificationRequest("<CHANNELS>", "<TEMPLATES>", "<EMAILS>", "<SEARCH>");
    val notifications = teleflow.notifications(notificationRequest)
    println(notifications)
}
```
