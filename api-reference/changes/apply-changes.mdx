
```php PHP
use Teleflow\SDK\Teleflow;

$teleflow = new Teleflow(<TELEFLOW_API_KEY>);
$teleflow->applyBulkChanges([
    'changeIds' = [
        '<insert-all-the-change-ids>'
    ]
])->toArray();
```

```ruby Ruby
require 'teleflow'

client = Teleflow::Client.new('<TELEFLOW_API_KEY>')

client.apply_change({
    'changeIds' => ['<insert-all-the-change-ids>']
})
```

```python Python
from teleflow.api import ChangeApi

url = "https://api.teleflow.khulnasoft.com"
api_key = "<TELEFLOW_API_KEY>"
teleflow = ChangeApi(url, api_key).bulk_apply(
    change_ids = "<The list of IDs to apply>",
)
```

```java Java
import co.teleflow.sdk.Teleflow;

public class Main {
    public static void main(String[] args) {
        String apiKey = "<TELEFLOW_API_KEY>";

        Teleflow teleflow = new Teleflow(apiKey);
        teleflow.applyBulkChanges("<list-of-change-ids>");
    }
}
```
